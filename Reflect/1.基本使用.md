# Reflect 
反射让我们能在运行期探知对象的类型信息和内存结构。

这样弥补了静态语言在行为上的不足。

```go
package main

import (
	"reflect"
	"fmt"
)

type user struct {
	name string
	age  int
}
type manager struct {
	user
	title string
}

func main() {
	var m manager
	t := reflect.TypeOf(&m)

	if t.Kind() == reflect.Ptr {
		t = t.Elem()
	}

	for i := 0; i < t.NumField(); i++ {
		f := t.Field(i)
		fmt.Println(f.Name, f.Type, f.Offset)

		if f.Anonymous {
			fmt.Println("Anonymouse~~")
			for x := 0; x < f.Type.NumField(); x++ {
				af:=f.Type.Field(i)
				fmt.Println(af.Name,af.Type)
			}

		}
	}
}
```
